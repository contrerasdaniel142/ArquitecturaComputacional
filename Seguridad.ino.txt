void color (unsigned char red, unsigned char green, unsigned char blue) // the color generating function
{
analogWrite(LED_RED, red);
analogWrite(LED_BLUE, blue);
analogWrite(LED_GREEN, green);
}
void inpass(){
key = keypad.getKey(); 
lcd.setCursor(0,0);
lcd.println("Contrasenia");
  Tiempo.Update();
  if (key){     
    
    lcd.setCursor(i, 1);
    if(key!='*'&&i<5){
      lcd.println('*');
      Input[i]=key;
      i++;
    }
    else{   
      lcd.clear();
      lcd.setCursor(0, 1);  
      i=0;
      Serial.println(Input);
      Serial.println(Password);
      if(strncmp(Password,Input,5)==0){
        cont=5;        
      }else{
        lcd.println("Incorrecta");
        cont++;
        color(0,255,0);               
        delay(2000);
        lcd.clear();
                             
      }              
      lcd.setCursor(0, 0); 
    }
  }  
}
void temp(){
  key = '*';
}
void loop(){
  if(cont<3){
    Clave.Update();
  }else if(cont == 3){
    lcd.println("Bloqueado");
    for (int i = 0; i < 25; i++) {			// bucle de 25 iteraciones
          int duracion = 1500 / duraciones[i];		// duracion de la nota ms
          tone(BUZZER, melodia[i], duracion);	// ejecuta el tono con la duracion
          int pausa = duracion * 1.30;			// calcula pausa
          delay(pausa);					// demora con valor de pausa
          noTone(BUZZER);          
        }
        cont=4;
  }else if(cont == 5){
        lcd.println("Correcta");       
        color(255,0,0);  
        cont=4;  
  }
}
